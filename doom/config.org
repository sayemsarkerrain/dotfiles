*** Templates
#+begin_src emacs-lisp
(setq org-capture-templates
      '(
        ;; Todo Capture
        ("t" "tasks" entry
         (file ,(concat org-directory "tasks.org"))
         "* TODO %?\n:PROPERTIES:\n:date: %U\n:END:\n")

        ;; Idea Capture
        ("i" "idea"entry
         (file ,(concat org-directory "ideas.org"))
         "* IDEA %? :ideas:\nPROPERTIES:\n:date: %U\n:END:\n")

        ;; Note/Log Capture
        ("j" "journal" entry
         (file+olp+date+tree ,(concat org-directory "journal.org"))
         "* %U\n%?\n"
        )))

#+end_src
